uniCloud/cloudfunctions/user-info/index.jsNode.js
用户云函数：支持获取个人资料与更新实名信息，确保 userId 隔离。
'use strict';
const db = uniCloud.database();
exports.main = async (event) => {
  const { userId, action, data } = event;
  const col = db.collection('user');
  if (action === 'get') {
    const res = await col.where({ userId }).get();
    return { code: 200, data: res.data[0] };
  }
  if (action === 'update') {
    await col.where({ userId }).update({ ...data, updateTime: Date.now() });
    return { code: 200, msg: '更新成功' };
  }
};
